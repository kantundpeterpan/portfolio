Simulated dataset

The data will be simulated counts a multinomial distribution whose parameter vector \(p_j\) depends on the switchpoint \(s\):

\[ \begin{aligned} (X_i \mid p(t)) &\sim \mathcal{Mult}(n, p(t)) \\ &s \sim Uniform(t_0, t_i) \\ &\text{with }p_j(t) = \left[p_{[j,1]}, \ldots, p_{[j,k]}\right] = \begin{cases} p_1 \mid t<s \\ p_2 \mid t>s \end{cases} \\ p_1 &\sim Dir(\alpha_1 = \{\alpha_{1,1}, \ldots, \alpha_{1,j} \}) \\ p_2 &\sim Dir(\alpha_2 = \{\alpha_{2,1}, \ldots, \alpha_{2,j} \}) \end{aligned} \]

Model specification

idx = np.hstack([np.arange(0,len(data)).reshape(-1,1)]*4)
n_samples_prior = int(10 / n_counts)
# Prior for the Dirichlet distribution
alpha_prior = tf.constant(1.0, shape=(4,))  # Uninformative prior
def model(p_fn):
    ddist1 = tfd.Dirichlet(alpha_prior,
                           force_probs_to_zero_outside_support=False)
    
    p1 = tfd.Sample(ddist1)
    
    ddist2 = tfd.Dirichlet(alpha_prior,
                       force_probs_to_zero_outside_support=False)

    p2  = tfd.Sample(ddist2)
    
    s = tfd.Uniform(0., high=len(data))
    
    x_i = lambda s, p2, p1: tfd.Independent(
        tfd.Multinomial(total_count= data.sum(axis = 1),
                        probs = p_fn(idx, s, p2, p1),
                        #reinterpreted_batch_ndims=1
                       )
    )
    
    outcome = tfd.JointDistributionNamedAutoBatched(
        dict(
            #ddist1 = ddist1,
            p1 = p1,
            #ddist2 = ddist2,
            p2 = p2,
            s = s,
            x_i = x_i
            ),
    )

    return outcome
from functools import partial
def p_switch(ys, s, p2, p1):
    return tf.where(ys < s, tf.stack([p1]*len(data), axis = 0), tf.stack([p2]*len(data), axis=0))
pp_switch = partial(p_switch, ys = idx)
def p_sigmoid(ys, s, p2, p1):
    #ys = np.hstack([np.arange(0,len(ys)).reshape(-1,1)]*4)
    return p1 + tf.sigmoid(ys - s) * (p2 - p1)
# two models with different switch functions
model_switch = model(p_switch)
model_sigmoid = model(p_sigmoid)
def target_log_prob_fn(model, s, p1, p2):
  return model.log_prob(s=s, p1=p1, p2=p2, x_i=data)
